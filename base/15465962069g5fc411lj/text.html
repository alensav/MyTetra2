# скрипт синхронизации data на ubuntu

sudo nano /etc/init.d/Sync_otkl_data_Time


sudo nano /etc/gdm/PostSession/Default

#та команда создаст по 2 ссылки в каталогах /etc/rc0.d (второе число в команде) и /etc/rc6.d (третье число в команде). Причём вначале будет выполняться сценарий с параметром stop (т.к. стоит 1), а уже потом с параметром start (т.к. стоит 20).
Если второй параметр не нужен, то можно выполнить команду

sudo update-rc.d /etc/init.d/Sync_otkl_data_Time stop 1 0 6 .
