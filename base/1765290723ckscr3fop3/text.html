Профилактика «диска заполнен, но где?»

==========================================

# 1. Ищем удалённые, но открытые файлы

sudo lsof +L1


# 2. Ищем большие файлы, скрытые за точками монтирования

sudo du -xh --max-depth=1 / 2>/dev/null | sort -hr


# 3. Проверяем journalctl (часто съедает ГБ)

journalctl --disk-usage

# Очистить: journalctl --vacuum-size=100M


# 4. Проверяем Docker (если используется)

docker system df

docker system prune -af # осторожно!


# 5. Ищем большие логи

find /var/log -type f -name "*.log" -size +100M -exec ls -lh {} \;
